二维数组

```
public class Main {
    public  static void main(String[] agrs){
        //二维数组声明和初始化
        int [] arr = new int[]{1,2,3};
        int [][] arr1 = new int[][]{{1,2,3}, {4,5},{6,7,8}};
        //声明时，不得声明长度和初始化值一同出现 ，
//        int [][] arr = new int[3][2]{{1,2,3},{1,2}};  //错误

        int [] arr4 = {1,2,3,4,5,5}; //类型推断

        /*
            数组默认初始化值
            二维数组，
            外层数组初始化值：地址值
            内层元素初始化值，与一维数组相同


         */
        int [][] arr5 = new int [4][3];
        System.out.println(arr5[0]);    //地址值 ： [I@372f7a8d

        /*
               数组的内存解析
               二维数组声明后，
               堆中生成内存地址，同时开辟固定长度，默认值为类型的默认值
               同时这个栈变量指向这个内存地址，此为外层数组
               其次，堆生成内存地址，开辟固定长度，默认值为变量类型的默认值
               同时堆中的内存地址数据，指向内存地址

         */



    }
}
```
